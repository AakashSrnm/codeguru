*Write A C Program to find the area and circumference of a circle*

#include<stdio.h>
main() {
    float radius, area, perimeter;
    printf("Enter the Radius of Circle:");
    scanf("%f", &radius);
    area = 3.14 * radius * radius;
    perimeter = 2 * 3.14 * radius;
    printf("\n\nThe Area Of the Circle with the radius %f", radius, area);
    printf("\n\nThe Circumference of the same circle is %f", perimeter);
    getch();
}

*Write a C program to find the greatest of 3 numbers*

#include<stdio.h>
#include<conio.h>
int main() {
    int a, b, c;
    printf("Enter Three Numbers:");
    scanf("%d%d%d", &a, &b, &c);
    if (a > b && a > c) {
        printf("%d is Largest", a);
    }
    if (b > a && b > c) {
        printf("%d is Largest", b);
    }
    if (c > a && c > b) {
        printf("%d is Largest", c);
    }
    if (a == b && a == c) {
        printf("All Are Equal");
    }
    return 0;
}
